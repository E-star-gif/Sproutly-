<script>
let plants = JSON.parse(localStorage.getItem("plants") || "[]");

function save() {
  localStorage.setItem("plants", JSON.stringify(plants));
}

/* ---------------- NOTIFICATIONS ---------------- */

function requestNotif() {
  if (!("Notification" in window)) return;

  Notification.requestPermission().then(p => {
    alert(p === "granted"
      ? "Sproutly notifications enabled 🌿"
      : "Notifications blocked");
  });
}

function notify(title, body) {
  if (Notification.permission === "granted") {
    new Notification(title, { body });
  }
}

/* periodic check */
setInterval(checkPlants, 60000);

function checkPlants() {
  plants.forEach(p => {
    const days = daysSince(p.lastWatered);
    if (days >= p.frequency) {
      notify("🌿 Sproutly", `${p.name} needs water`);
    }
  });
}

/* ---------------- CORE ---------------- */

function addPlant() {
  const name = document.getElementById("name").value;
  const freq = document.getElementById("frequency").value;

  if (!name) return;

  plants.push({
    id: Date.now(),
    name,
    frequency: parseInt(freq),
    lastWatered: Date.now()
  });

  save();
  render();
  document.getElementById("name").value = "";
}

function waterPlant(id) {
  plants = plants.map(p => {
    if (p.id === id) p.lastWatered = Date.now();
    return p;
  });
  save();
  render();
}

function deletePlant(id) {
  plants = plants.filter(p => p.id !== id);
  save();
  render();
}

function daysSince(time) {
  return Math.floor((Date.now() - time) / (1000*60*60*24));
}

/* ---------------- UI ---------------- */

function render() {
  const el = document.getElementById("plants");
  el.innerHTML = "";

  if (plants.length === 0) {
    el.innerHTML = "<p>No plants yet 🌱 Add your first one above.</p>";
    return;
  }

  plants.forEach(p => {
    const days = daysSince(p.lastWatered);
    const due = days >= p.frequency;

    const div = document.createElement("div");
    div.className = "card";

    div.innerHTML = `
      <div class="plant">
        <div>
          <strong>${p.name}</strong><br/>
          <span class="small">Water every ${p.frequency} days</span><br/>
          <span class="${due ? "danger" : "ok"}">
            ${due ? "⚠ Needs water" : "✔ Healthy"}
          </span>
        </div>
      </div>

      <button onclick="waterPlant(${p.id})">💧 Mark Watered</button>
      <button onclick="deletePlant(${p.id})" style="background:#e74c3c;">Delete</button>
    `;

    el.appendChild(div);
  });
}

render();
